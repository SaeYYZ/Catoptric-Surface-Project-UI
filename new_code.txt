create table
  public."DashboardSelection"  (
    id bigint generated by default as identity not null,
    created_at timestamp with time zone null default now(),
    user_id uuid null,
    mirror_selection int2[] null,
    constraint DashboardSelection_pkey primary key (id)
  ) tablespace pg_default;



create table public."Mirrors" (
    id bigint generated by default as identity not null,
    created_at timestamp with time zone null default now(),
    top_left int2[] null,
    left_most int2[] null,
    right_most int2[] null,
    top_right int2[] null,
    bottom_left int2[] null,
    bottom_right int2[] null,
    user_id uuid null,
    constraint Mirrors_pkey primary key (id)
  ) tablespace pg_default;



create table public."Prototype" (
    id bigint generated by default as identity not null,
    created_at timestamp with time zone null default now(),
    top_left json null,
    top_right json null,
    bottom_left json null,
    bottom_right json null,
    user_id uuid null,
    constraint Prototype_pkey primary key (id)
  ) tablespace pg_default;
